### 43. Lookup Table

- key-value로 관리되는 배열이 나열된 형태

```javascript
function getUserType(type) {
    const USER_TYPE = {
        ADMIN: '관리자',
        INSTRUCTOR: '강사',
        STUDENT: '수강생',
        UNDEFINED: '해당 없음',
    };
    
    return USER_TYPE[type] ?? USER_TYPE[USER_TYPE.UNDEFINED];
};
    
getUserType('ADMIN');
    
    
function getUserType(type) {
    return ({
        ADMIN: '관리자',
        INSTRUCTOR: '강사',
        STUDENT: '수강생',
    }[type] ?? '해당 없음);
}
 
getUserType('ADMIN');
```
